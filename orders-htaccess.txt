RewriteEngine On

# Handle OPTIONS requests
RewriteCond %{REQUEST_METHOD} OPTIONS
RewriteRule ^(.*)$ $1 [R=200,L]

# Route order detail requests to order-detail.php
RewriteRule ^(\d+)$ order-detail.php?id=$1 [L,QSA]
RewriteRule ^([a-zA-Z0-9-]+)$ order-detail.php?id=$1 [L,QSA]

# Allow direct access to PHP files
RewriteCond %{REQUEST_FILENAME} -f
RewriteRule ^.*\.php$ - [L]
